```python
import pandas as pd
import os

def load_all_provinces(folder):
    dfs = []

    for file in os.listdir(folder):
        if file.endswith(".xlsx"):
            df = pd.read_excel(os.path.join(folder, file))

            df['province'] = file.replace(".xlsx","")
            df['date'] = pd.to_datetime(df['year'].astype(str) + '-' + df['month'].astype(str) + '-01')

            dfs.append(df)

    df_all = pd.concat(dfs)
    df_all = df_all.sort_values(['province','date'])

    df_all = df_all.fillna(method='ffill').fillna(method='bfill')

    return df_all
```

```python
import torch
import torch.nn as nn

class LSTMModel(nn.Module):
    def __init__(self, input_size):
        super().__init__()
        self.lstm = nn.LSTM(input_size, 64, batch_first=True)
        self.fc = nn.Linear(64, 3)  # 3 horizons

    def forward(self, x):
        out,_ = self.lstm(x)
        out = out[:,-1,:]
        return self.fc(out)
```

## 8️⃣ `trainer.py`

```python
import torch
import numpy as np

def train_lstm(model, X, y, epochs=10):
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    loss_fn = torch.nn.MSELoss()

    for epoch in range(epochs):
        model.train()
        pred = model(X)
        loss = loss_fn(pred, y)

        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

    return model
```

## 1️⃣ `src/shap_analysis.py`

```python
import shap
import pandas as pd
import numpy as np
import os
import matplotlib.pyplot as plt

def run_shap_analysis(model, X_train, X_test, feature_names, output_dir):
    os.makedirs(output_dir, exist_ok=True)

    # SHAP explainer
    explainer = shap.Explainer(model, X_train)

    shap_values = explainer(X_test)

    # =========================
    # GLOBAL FEATURE IMPORTANCE
    # =========================
    shap.summary_plot(shap_values, X_test, feature_names=feature_names, show=False)
    plt.savefig(f"{output_dir}/shap_summary.png")
    plt.close()

    # =========================
    # SAVE MEAN ABS SHAP
    # =========================
    shap_importance = np.abs(shap_values.values).mean(axis=0)

    df_importance = pd.DataFrame({
        "feature": feature_names,
        "importance": shap_importance
    }).sort_values(by="importance", ascending=False)

    df_importance.to_csv(f"{output_dir}/top_features.csv", index=False)

    return df_importance
```

## 2️⃣ SHAP theo tỉnh (QUAN TRỌNG)

```python
def shap_by_province(model, df_test, feature_cols, province_col, output_dir):
    import shap
    import numpy as np
    import pandas as pd

    results = []

    for province in df_test[province_col].unique():
        df_p = df_test[df_test[province_col] == province]

        if len(df_p) < 10:
            continue

        X_p = df_p[feature_cols]

        explainer = shap.Explainer(model)
        shap_values = explainer(X_p)

        mean_shap = np.abs(shap_values.values).mean(axis=0)

        for f, val in zip(feature_cols, mean_shap):
            results.append({
                "province": province,
                "feature": f,
                "importance": val
            })

    df_out = pd.DataFrame(results)
    df_out.to_csv(f"{output_dir}/shap_by_province.csv", index=False)

    return df_out
```

```python
import pandas as pd

def generate_insights(top_features_path, shap_by_province_path, output_path):
    top_df = pd.read_csv(top_features_path)
    prov_df = pd.read_csv(shap_by_province_path)

    insights = []

    # GLOBAL TOP FEATURES
    top3 = top_df.head(3)["feature"].tolist()
    insights.append(f"Top 3 important features globally: {top3}")

    # WEATHER VS DISEASE
    weather_features = [f for f in top3 if "Rain" in f or "Humidity" in f or "temperature" in f]

    if weather_features:
        insights.append(f"Climate variables significantly influence prediction: {weather_features}")

    # LAG DETECTION
    lag_features = [f for f in top3 if "lag" in f]
    insights.append(f"Lagged effects detected in: {lag_features}")

    # REGIONAL DIFFERENCE
    grouped = prov_df.groupby("feature")["importance"].std().sort_values(ascending=False)

    if len(grouped) > 0:
        top_var = grouped.index[0]
        insights.append(f"Feature {top_var} shows strongest regional variation")

    # SAVE
    with open(output_path, "w") as f:
        for line in insights:
            f.write(line + "\n")

    return insights
```
